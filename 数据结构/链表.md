# 链表

## 特点

        链表存储有序的元素集合，但不同于数组，链表中的元素在内存中并不是连续放置的。每个元素由一个存储元素本身的节点和一个指向下一个元素的引用（也称指针或链接）组成。 
        相对于传统的数组，链表的一个好处在于，添加或移除元素的时候不需要移动其他元素。然 而，链表需要使用指针，因此实现链表时需要额外注意。数组的另一个细节是可以直接访问任何 位置的任何元素，而要想访问链表中间的一个元素，需要从起点（表头）开始迭代列表直到找到 所需的元素

function LinkedList() {
    let Node = function(element) {
        this.element = element
        this.next = null
    }
    // 存储链表的长度
    let length = 0
    // 存储头结点的地址
    let head = null

    // 向列表尾部添加一个新的项
    this.append = function(element) {
        let node = new Node(element)
        // 保存当前节点
        let current = null
        if (head === null) {
            head = node
        } else {
            current = head
            while(current.next) {
                current = current.next
            }
            current.next = node
        }
        length ++
    }
    // 根据给定位置移除一个元素
    this.removeAt = function(position) {
        // 因为对链表进行操作都需要遍历链表，所以用current保存当前节点
        let current = head
        // 保存上一节点，因为删除，某个节点本质上就是将某个节点的上一节点的next保存为某个节点的下一节点
        let previous = null
        let index = 0
        // 判断边界情况，要形成一种严谨的思维
        if (position > -1 && position < length) {

                head = current.next
